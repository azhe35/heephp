# 不存在记录

NotEXISTS()是条件语句，参数为一个SQL语句\

需配合value()、select()、get()、all()使用

示例：

//如果商品属性Id=1不存在，那么返回所有商品属性记录
$list = table('product_attr')->NotEXISTS(function(orm $bulider){
                                    return $bulider->table('product')->where("product_attr_id=1")->sql();
                                })->all();


//如果1班不存在，那么返回所有学生记录
$list = table('student')->NotEXISTS(function(orm $bulider){
                                    return $bulider->table('`class`')->where("`name`='1班'")->sql();
                                })->all();
