    # 查询条件

    # where

    设置查询条件，参数1个：条件字符串、函数、数组

    示例：

    //使用字符串查询
$list = table('users')->where('id=1')->select();

//使用数组查询
$list = table('users')->where(['id'=>1])->select();

//使用回调函数查询
$list = table('users')
        ->where(function(wherebulider $wb){
            return $wb->where('id=1')->sql();
        })->select();

    # whereAnd

    设置AND连接查询条件，参数1个：条件字符串、函数、数组

    示例：

    //使用字符串查询
$list = table('users')->where('id>0')->whereAnd("name like '%a%'")->select();

//使用数组查询
$list = table('users')->where("id>0")->whereAnd(['state'=>1,'group'=>'用户组'])->select();

//使用回调函数查询
$list = table('student')
        ->where('score>=60')
        ->whereAnd(function(wherebulider $wb){
            return $wb->where("city='北京'")
                      ->whereAnd("class='1班'")
                      ->sql();
        })->select();

    # whereOr

    设置OR连接查询条件，参数1个：条件字符串、函数、数组

    示例：

    //使用字符串查询
$list = table('users')->where('id>0')->whereOr("name like '%a%'")->select();

//使用数组查询
$list = table('users')->whereOr(['state'=>1,'group'=>'用户组'])->select();

//使用回调函数查询
$list = table('student')
        ->where('score>=60')
        ->whereOr(function(wherebulider $wb){
            return $wb->where("city='北京'")
                      ->whereAnd("score>50")
                      ->sql();
        })->select();

    # whereIn

    查询字段存在值列表中的记录，参数2个：字段名、条件字符串、函数、数组

    示例：

    //使用字符串查询
$list = table('users')->where('id>0')->whereIn("city in ('上海'，'北京')")->select();

//使用数组查询
$list = table('users')->whereIn('id',[1,3,5,6,7])->select();

//使用回调函数查询
$list = table('student')
        ->where('score>=60')
        ->whereIn(function(wherebulider $wb){
            $list =[1,3,5,6,7];
            return $list
        })->select();

    # whereNotIn

    查询字段不存在值列表中的记录，参数2个：字段名、条件字符串、函数、数组

    示例：

    //使用字符串查询
$list = table('users')->where('id>0')->whereNotIn("city in ('南京'，'长沙')")->select();

//使用数组查询
$list = table('users')->whereNotIn('id',[10,11,12])->select();

//使用回调函数查询
$list = table('student')
        ->where('score>=60')
        ->whereNotIn(function(){
            $list =[10,11,12];
            return $list
        })->select();

    # whereBetween

    查询字段值介于两个值之间的记录，参数2个：字段名、值1、值2

    示例：

    //查询年龄在18-28之间的用户
$list = table('users')->where('id>0')->whereBetween('age',18,28)->select();

//查询成绩介于60-70之间的学生
$list = table('student')->whereBetween('score',60,70)->select();

    # whereNotBetween

    查询字段值不在两个值之间的记录，参数2个：字段名、值1、值2

    示例：

    //查询年龄不在0-18之间的用户
$list = table('users')->where('id>0')->whereNotBetween('age',0,18)->select();

//查询成绩不存在0-59之间的学生
$list = table('student')->whereNotBetween('score',0,59)->select();

    # whereIsNULL

    查询字段值是NULL的记录，参数1个：字段名

    示例：

    //查询城市为NULL的用户
$list = table('users')->whereIsNULL('city')->select();

//查询成绩为NULL的用户
$list = table('student')->whereIsNULL('score')->select();

    # whereIsNotNULL

    查询字段值不是NULL的记录，参数1个：字段名

    示例：

    //查询城市不为NULL的用户
$list = table('users')->whereIsNotNULL('city')->select();

//查询成绩不为NULL的用户
$list = table('student')->whereIsNotNULL('score')->select();

    # whereEmpty

    查询字段值是NULL、空字符串、0的记录，参数1个：字段名

    示例：

    //查询所有父类别
$list = table('category')->whereEmpty('parent_id')->select();

//查询成绩为空的用户
$list = table('student')->whereEmpty('score')->select();
