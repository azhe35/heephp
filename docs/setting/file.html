# 配置文件

系统有一个总的配置文件为\app\config.php

每个应用的根目录可有一个配置文件位于：\app\{home}\config.php

应用目录的配置项会覆盖系统配置的配置项，如果应用目录的配置文件没有配置项，则会使用系统总配置的配置项

用户可自行增加配置项，格式为 PHP 的数组 key=>value键值对

# 配置示例
return [
    'default_app'=>'index',//默认APP 启用多应用时有效
    'default_controller'=>'index',//默认的控制器
    'default_method'=>'index',//默认的方法
    'debug'=>true,//是否开启debug模式
    'logger'=>true,//是否开启日志
    'pagesize'=>5,//系统分页数量
];

可使用二级配置

# 配置实例
return [
    'db'=>[//数据库配置项
            'diver'=>'mysqli',//数据库驱动 目前支持mysqli PDO
            'db_host' => 'localhost',//数据库服务器
            'db_port' => '3306',//数据库端口
            'db_username' => 'root',//数据库用户名
            'db_password' => '',//数据库密码
            'db_name' => '',//数据库名称
            'table_prefix'=>''//数据表前缀
    ]
];
