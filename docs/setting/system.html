# 系统配置

以下为默认系统配置项目详解

return [
    'default_app'=>'index',//默认的APP 开启多应用模式有效
    'default_controller'=>'index',//默认的控制器
    'default_method'=>'index',//默认的方法名
    'debug'=>true,//是否开启debug模式
    'logger'=>true,//是否开启日志
    'pagesize'=>5,//分页数量
    'format_suffix'=>'htm',//URL路由后缀
    'validata_error_page'=>'message/validataError.php',//验证错误的提示页模板
    'success_page'=>'message/sysSuccess.php',//操作成功的提示页模板
    'error_page'=>'message/sysError.php',//操作失败的提示页模板
    'validata_code_session'=>'_validata_code_session_',//验证码的Session名称
    'db'=>[//数据库配置
        'diver'=>'mysqli',//数据库驱动 目前支持mysqli PDO ，mysql数据库建议使用mysqli ， PDO支持mysql sqlite数据库
        'db_host' => 'localhost',//数据库服务器
        'db_port' => '3306',//数据库端口
        'db_username' => 'root',//数据库登录名
        'db_password' => '',//数据库登录密码
        'db_name' => '',//数据库名
        'table_prefix'=>'',//数据表前缀
        'charset'=>'utf8',//数据库的默认编码
        'timeformat'=>'Y-m-d H:i:s',//读取时间的格式
        'dateformat'=>'Y-m-d',//读取日期的格式

    ],
    'pagination'=>[//分页模板的class类名 默认采用bootstrap的类定义
      'class'=>'pagination',
        'item_class'=>'page-item',
        'link_class'=>'page-link',
        'currt_class'=>'active',
    ],
    'lang'=>[//多语言的配置
        'on'=>false,//是否开启多语言
        'default'=>'zh-cn'//默认的语言
    ],
    'cache'=>[//缓存
        'diver'=>'file',//缓存的类型 目前支持 file  redis  memcache
        'exp_time'=>3600,//缓存的超时时间
        'redis'=>[//diver为redis时有效
            'host'=>'127.0.0.1',//redis服务器地址
            'port'=>'6379'//redis服务器端口
        ],
        'memcache'=>[//diver为memcache时有效
            'host'=>'127.0.0.1',//memcache服务器地址
            'port'=>'11211'//memcache服务器端口
        ]
    ]
];
