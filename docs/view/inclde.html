    # 包含文件

    模板之间的包含建议使用 HeePHP 提供的函数 import()方法。
    被引用的文件，可直接使用控制器中设置的变量

    //包含view根目录的layout中header.php
import('/layout/header.php');
//包含与当前模板文件同级的sider.php
import('sider.php');
//包含与当前模板文件子目录setting目录中的basic.php
import('setting/basic.php');

    模板使用示例：

    <html>
  <body>
    <header>
      <?import('/layout/header.php')?>
    </header>
    <div class="sider">
      <?import('sider.php')?>
    </div>
    <div>
      <?import('setting/baisc.php')?>
    </div>
    <footer>
      版权所有 2020
    </footer>
  </body>
</html>

    # 为包含文件单独设置变量

    //变量赋值
import('sider.php',['title'=>'标题','item'=>'项目']);


    标题：<?=$title?>
项目：<?=$item?>
