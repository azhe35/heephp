    # 读取数据

    读取数据是常用的操作，模型提供了以下常用操作：

    读取后的数据会被存入 模型的 data 中，可通过{$model}->data 获取

    # 单条记录方法：

    方法名 参数 参数类型 返回 说明
    find where string 第一条记录 或 false 根据条件返回一条记录
    get id string 一条记录 或 false 获取主键值，返回一条记录
    //返回用户名密码正确的第一条记录
$u = $users->find("username='$username' and password='md5($password)'");

//返回Id为1的用户
$u=$users->get(1)

    # 多条记录方法：

    select方法获取多条记录，其参数都不是必须的，参数列表如下：

    当参数为空时获取全部数据，select方法获取的数据不带有分页信息,需要使用分页的使用page方法

    参数 默认值 是否必须 类型 说明
    where 空 否 string where 条件
    order 空 否 string 排序的字符串
    fields '*' 否 string 要获取的字段列表
    onlysoftdel false 否 bool 是否仅取出被软删除的数据
    pname 'page' 否 string page 分页在 url 中的字符串
    page 0 否 int 要获取数据的页码
    //获取所有数据
$list = $meuns->select();

//根据条件获取数据
$list = $users->select('age<18');

//根据条件获取数据并排序
$menus->select("parent_id=0",'ord desc');
//从模型的data中获取数据
$list = $menus->data;

//仅获取软删除的数据
$list = $menus->select('','','*',true);

//仅获取第2页的数据  pagesize在配置文件\app\config.php中
$list = $users->select('','','*',false,'page',2);

    page方法获取多条记录，其参数都不是必须的，参数列表如下：

    当参数为空时获取第一页的数据

    参数 默认值 是否必须 类型 说明
    where 空 否 string where 条件
    order 空 否 string 排序的字符串
    fields '*' 否 string 要获取的字段列表
    onlysoftdel false 否 bool 是否仅取出被软删除的数据
    pname 'page' 否 string page 分页在 url 中的字符串
    //获取当前页的数据 页码根据pname参数从url中自动读取 默认为第一页
$list = $users->page();

//根据条件获取年龄大于8岁的并分页  pagesize在配置文件\app\config.php中
$list = $users->page('age>8');

//根据条件获取数据并排序
$list = $menus->page("parent_id=0",'ord desc');

//仅获取软删除的数据
$menus->page('','','*',true);
//从模型的data中获取数据
$list = $menus->data;

//当页面有多个需要分页的数据列表时，需要变更pname的值
$users->page('age>8','id desc','*',false,'pusers');
//从模型的data中获取数据
$list = $menus->data;
//获取分页信息
$pager = $users->pager;
//获取分页的代码
$pshow = $users->pager['show'];

    # 其他获取数据的方法

    方法名 参数 参数类型 返回说明
    getBy 字段名 where string null 或 值 或 字段名值数组
    
//获取users_group_id为1的用户的username的数组 为一个时：仅返回其username字符串
$username_list =$users->getByusername('user_group_id=1');

//返回用户名包含admin字符串的id列表
$ids = $users->getByid('username like "%admin%"')
