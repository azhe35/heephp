    # 对多关联

    对多关系调用仅需要两行代码：

    $relation = new relation($this,关联模型,外键,关联模型的关联字段);
    return $relation->hasmore(true|false);

    示例：

    class menus extends heephp\model{

    //获取子菜单
    public function child(){
        //创建关联模型
        $relation = new relation($this,'menus',$this->key,'parent_id','child');
        //设置关联模型需要调用的方法
        $relation->set_rmodel_methods('create_user');
        //返回关联模型列表 参数为是否分页
        return $relation->hasmore(false);
    }

}

    调用：

    //创建模型
$menus=model('menus');
//获取数据
$menus->select('parent_id=0');
//从关联模型中获取数据
$menus->child();

//获取最终数据
$list = $menus->data;



//示例二

//创建模型
$menus=model('menus');
//获取数据
$menus->get(1);
//从关联模型中获取数据
$menus->child();

//获取最终数据
$m = $menus->data;

    # relation 参数

    创建关联模型参数列表：

    参数名 类型 说明
    model heephp\model 当前模型，直接传入$this
    rtable string 关联模型的表名
    foreign_key string 当前模型的外键
    relation_field string 关联模型表的关联字段
    alias string 别名，参数可选

    hasmore 参数列表：

    参数名 类型 说明
    splitpage bool 是否分页（默认 false）
