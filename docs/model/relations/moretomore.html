    # 多对多关联

    多对多关系调用仅需要两行代码：

    $relation = new relation($this,关联模型,外键,关联模型的关联字段,中间表名);
    return $relation->moretomore(true|false);

    示例：

    class users_group extends heephp\model{

    //获取用户组关联资源列表
    public function resources(){

        //创建关联模型
        $relation = new relation($this,'sys_resources','users_group_id','resources_id','','users_group_resources');
        //返回关联模型列表 参数为是否分页
        return $relation->moretomore(false);

    }

}

    调用：

    //创建模型
$users_group=model('users_group');
//获取数据
$users_group->select();
//从关联模型中获取数据
$users_group->resources();

//获取最终数据
$list = $users_group->data;

    # relation 参数

    创建关联模型参数列表：

    参数名 类型 说明
    model heephp\model 当前模型，直接传入$this
    rtable string 关联模型的表名
    foreign_key string 当前模型的外键
    relation_field string 关联模型表的关联字段
    alias string 别名，参数可选
    midtalbe string 中间表名

    hasmore 参数列表：

    参数名 类型 说明
    splitpage bool 是否分页（默认 false）
