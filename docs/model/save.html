    # 保存数据

    数据保存为常用操作，因此内置两个常用的方法：

    插入和更新的 lastid 不会被存入模型的 data 中，可从返回的数据中获取

    # 插入数据

    //插入提交的数据
$data=request('post.');
$users=model('users');
$lastid = $users->insert($data);

//插入数据
$data['name']='李明';
$data['age']=8;
$data['school']='一小';
$lastid = $users->insert($data);

//插入数组数据
$lastid = $users->insert(['name'=>'李明','age'=>'8','school'=>'一小']);

    # 修改数据

    //修改提交的数据 $data中需要有主键关联的值
$data=request('post.');
$users=model('users');
$affect_row = $users->update($data);

//修改数据 主键为id
$data['id']=1;
$data['name']='李明';
$data['age']=8;
$data['school']='一小';
$affect_row = $users->update($data);

//修改数组数据
$affect_row = $users->update(['id'=>1,'name'=>'李明','age'=>'8','school'=>'一小']);

//根据条件修改
$data['name']='李明';
$data['age']=8;
$data['school']='一小';
$affect_row = $users->where('id=1')->update($data);

//根据条件修改
$affect_row = $users->where('id=1')->update(['name'=>'李明','age'=>'8','school'=>'一小']);
