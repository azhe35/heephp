    # 请求

    请求是 web 应用的基本操作，因此 HeePHP 封装了请求功能

    统一使用request()函数，可进行方便的获取和设置。

    # GET 请求

    以下为 GET 请求的示例 GET 请求不存在时返回空字符串否则返回相应值

    //返回name值
$val = request('get.name');

//返回所有get请求的数组
$val = request('get.');

//返回age值
$val = request('GET.age');

//返回nickname值
$val = request('g.nickname');

//返回mobile值
$val = request('G.mobile');

    # POST 请求

    POST 请求不存在时返回空字符串否则返回相应值

    //返回name值
$val = request('post.name');

//返回所有get请求的数组
$val = request('post.');

//返回age值
$val = request('POST.age');

//返回nickname值
$val = request('p.nickname');

//返回post所有值的数组
$val = request('POST.');

    # SESSION

    SESSION 请求不存在时返回空字符串否则返回相应值

    //返回session.name值
$val = request('session.name');

//设置session.name的值为user1  并返回user1
$user = request('session.name','user1');

    # COOKIE

    COOKKIE 请求不存在时返回空字符串否则返回相应值

    //返回cookie.name值
$val = request('cookie.name');

//设置cookie.name的值为user1  并返回user1
$user = request('cookie.name','user1');

    # SERVER

    SERVER 请求不存在时返回空字符串否则返回相应值

    //返回浏览器用户的IP
$val = request('server.REMOTE_ADDR');

//返回当前服务的端口
$port = request('ser.SERVER_PORT');
